18     class ControllerStockKeeper implements ActionListener{
19         Aplikasi model;
20         StockKeeper view;
21     
22         public ControllerStockKeeper() {
23             view = new StockKeeper();
24             model = new Aplikasi();
25             view.setVisible(true);
26             view.setActionListener(this);
27             model.setTableRestock(view.getTableRestock());
28             model.setTableBarang(view.getTableBarang());
29         }
30     
31         public void actionPerformed(ActionEvent ae) {
32             Object source = ae.getSource();
33             Barang b = new Barang();
34             Restock re = new Restock();
35             
36             String nama = view.getTxNama();
37             int harga = view.getTxHarga();
38             int jumlah = view.getTextStock();
39             String status = view.getTxStatus();
40             int id = b.getId_barang();
41             
42             b = model.DataBarang(view.getTxSearch());
43             re = model.DataRestock(view.getTxSearch(),model.idRestock);
44             if(source.equals(view.getBtnSearch())){
45                 if(view.getTxSearch() == 0){
46                      JOptionPane.showConfirmDialog(view,"Data Tidak Ditemukan","ID Barang Salah", JOptionPane.WARNING_MESSAGE);
47                 } else{
48                     view.setTxIdBarang(b.getId_barang());
49                     view.setTxStock(b.getJumlah_barang());
50                     view.setTxNama(b.getNama_barang());
51                     view.setTxStock(b.getJumlah_barang());
52                     view.setTxStatus(b.getStatus());
53                     view.setTxHarga(b.getHarga_barang());
54                     view.setTxAdd(re.getJumlahRestock());
55                 }
56             } else if (source.equals(view.getBtnUpdate())){
57                 if(view.getTxIdBarang()==0 && view.getTxStock()==0){
58                     JOptionPane.showConfirmDialog(view,"Data Tidak ada", "Gagal Update",JOptionPane.WARNING_MESSAGE);
59                 } else{
60                     model.UpdateDataBarang(view.getTxIdBarang(), view.getTxNama(), (view.getTxStock() + view.getTxAdd()), view.getTxHarga());
61                     JOptionPane.showConfirmDialog(view, "Restock telah berhasil","Data TerUpdate",JOptionPane.WARNING_MESSAGE);
62                     model.setTableBarang(view.getTableBarang());
63                 }
64             } else if(source.equals(view.getBtnDelete())){
65                 model.DeleteBarang(view.getTxIdBarang());
66                 model.setTableBarang(view.getTableBarang());
67                 view.setVisible(false);
68                 new ControllerKasir();
69             } else if(source.equals(view.getBtnAdd())){
70                 if(( !nama.equals("") && harga!=0 && jumlah!=0 && status.equals(""))){
71                     JOptionPane.showConfirmDialog(view,"Input data berhasil", "Data telah ditambahkan",JOptionPane.WARNING_MESSAGE);
72                     model.tambahBarang(id, nama, jumlah, status, harga);
73                     view.reset();
74                 }
75             } else{
76                 view.setVisible(false);
77                 new ControllerLogin();
78             }
79         }
